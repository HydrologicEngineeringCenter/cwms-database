
  CREATE OR REPLACE FORCE VIEW "CWMS_20"."AV_SEC_TS_GROUP_MASK" ("DB_OFFICE_ID", "TS_GROUP_ID", "TS_GROUP_DESC", "TS_GROUP_MASK_DISPLAY", "DB_OFFICE_CODE", "TS_GROUP_CODE", "TS_GROUP_MASK") AS 
  SELECT	    c.office_id db_office_id,
                b.ts_group_id,
                b.ts_group_desc,
				CASE
                  WHEN a.ts_group_mask IS NULL
                  THEN
                    NULL
                  ELSE
                    cwms_util.denormalize_wildcards (a.ts_group_mask)
                  END ts_group_mask_display,
                db_office_code,
                ts_group_code,
				a.ts_group_mask
	  FROM		at_sec_ts_group_masks a
				RIGHT OUTER JOIN
					at_sec_ts_groups b
				USING (db_office_code, ts_group_code), cwms_office c
	 WHERE	db_office_code = c.office_code AND db_office_code != 0
